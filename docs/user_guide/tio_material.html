TIO_List_Material
PURPOSE
Returns the materials object asociated with the mesh
SYNOPSIS
C/C++:
 TIO_t TIO_List_Material( const TIO_File_t   fileID,
                          const TIO_Object_t meshID,
                          char               name[TIO_STRLEN] );
Fortran90
 integer(kind=TIO_ERRK) function TIO_List_Mat_f( fileID, meshID, name )

   integer(kind=TIO_FILEK), intent(in)  :: fileID
   integer(kind=TIO_OBJK),  intent(in)  :: meshID
   character(len=*),        intent(out) :: name
PARAMETERS
IN fileID TyphonIO file ID
IN meshID TyphonIO Mesh ID
OUT name The name of the TyphonIO Material
RETURNS
TyphonIO return code
DESCRIPTION


TIO_Create_Material
PURPOSE
Create a new material
SYNOPSIS
C/C++:
 TIO_t TIO_Create_Material( const TIO_File_t   fileID,
                            const TIO_Object_t meshID,
                            const char         name[],
                            TIO_Object_t       *materialID,
                            const TIO_Data_t   datatype,
                            const TIO_Size_t   nmat,
                            const TIO_Size_t   nghosts,
                            const TIO_Bool_t   ismixed,
                            const TIO_Data_t   ncomp_datatype,
                            const TIO_Data_t   ptr_datatype,
                            const TIO_Data_t   vf_datatype );
 ncomp_datatype and ptr_datatype must be set to TIO_TYPECLASS_SIGNED_INT, vf_datatype must be set to TIO_TYPECLASS_FLOAT.
Fortran90
 integer(kind=TIO_ERRK) function TIO_Create_Mat_f( fileID, meshID, name, materialID,
                                                   datatype, nmat, nghosts, ismixed,
                                                   ncomp_datatype, ptr_datatype, vf_datatype )

   integer(kind=TIO_FILEK), intent(in)           :: fileID
   integer(kind=TIO_OBJK),  intent(in)           :: meshID
   character(len=*),        intent(in)           :: name
   integer(kind=TIO_OBJK),  intent(out)          :: materialID
   integer(kind=TIO_IPK),   intent(in)           :: datatype
   integer(kind=TIO_SIZEK), intent(in)           :: nmat, nghosts
   logical(kind=TIO_LPK),   intent(in)           :: ismixed
   integer(kind=TIO_IPK),   intent(in), optional :: ncomp_datatype, ptr_datatype, vf_datatype
PARAMETERS
IN fileID TyphonIO file ID
IN meshID TyphonIO Mesh ID
IN name The material name
OUT materialID The material id
IN datatype TyphonIO data type
IN nmat Number of materials
IN nghosts Number of ghosts
IN ismixed Whether the material can handle mixed data
IN ncomp_datatype Datatype of num-component array
IN ptr_datatype Datatype of mat-pointer array
IN vf_datatype Datatype of volume-fraction array
RETURNS
TyphonIO return code
DESCRIPTION

 TIO_t TIO_Write_QuadMaterial_Chunk( const TIO_File_t   fileID,
                                     const TIO_Object_t materialID,
                                     const TIO_Size_t   idx,

                                     const TIO_Data_t   datatype,
                                     const void         *mat,
                                     const TIO_Data_t   ncomp_datatype,
                                     const TIO_Data_t   ptr_datatype,
                                     const TIO_Data_t   vf_datatype,
                                     const void         *ncomp,
                                     const void         *ptr,
                                     const void         *mixmat,
                                     const void         *vf );
 ncomp_datatype and ptr_datatype must be set to TIO_TYPECLASS_SIGNED_INT, vf_datatype must be set to TIO_TYPECLASS_FLOAT.
Fortran90
 integer(kind=TIO_ERRK) function TIO_Write_QuadMat_Chunk_f( fileID, materialID, idx, xfer,
                                                datatype, mat,
                                                ncomp_datatype, ptr_datatype, vf_datatype,
                                                ncomp, ptr, mixmat, vf  )

   integer(kind=TIO_FILEK),        intent(in)           :: fileID
   integer(kind=TIO_OBJK),         intent(in)           :: materialID
   integer(kind=TIO_SIZEK),        intent(in)           :: idx
   integer(kind=TIO_IPK),          intent(in)           :: xfer, datatype
   [INTEGER CHOICE], [DIMENSION],  intent(in)           :: mat
   integer(kind=TIO_IPK),          intent(in), optional :: ncomp_datatype, ptr_datatype, vf_datatype
   [INTEGER CHOICE], dimension(:), intent(in), optional :: ncomp, ptr, mixmat
   [REAL CHOICE],    dimension(:), intent(in), optional :: vf
PARAMETERS
IN fileID TyphonIO file ID
IN materialID TyphonIO material ID
IN idx Chunk index
IN xfer Data transfer method
IN datatype TyphonIO data type
IN mat Mesh material data array
IN ncomp_datatype Datatype of num-component array
IN ptr_datatype Datatype of mat-pointer array
IN vf_datatype Datatype of volume-fraction array
IN ncomp The mixed material num-component array
IN ptr The mixed material mat-pointer array
IN mixmat The mixed material mat-number array
IN vf The mixed material volumn fraction array
RETURNS
TyphonIO return code
DESCRIPTION

Fortran90 Notes:
[INTEGER CHOICE] must be the same for mat and mixmat, but others can differ.
